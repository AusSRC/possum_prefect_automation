server {
    if ($host = possum-prefect.aussrc.org) {
        return 301 https://$host$request_uri;
    } # managed by Certbot


    listen 80;
    server_name possum-prefect.aussrc.org;

    # Redirect HTTP → HTTPS
    return 301 https://$host$request_uri;


}

server {
    listen 443 ssl;
    server_name possum-prefect.aussrc.org;

    # SSL certs (use Let's Encrypt or your own)
    ssl_certificate /etc/letsencrypt/live/possum-prefect.aussrc.org/fullchain.pem; # managed by Certbot
    ssl_certificate_key /etc/letsencrypt/live/possum-prefect.aussrc.org/privkey.pem; # managed by Certbot

    ssl_protocols TLSv1.2 TLSv1.3;
    ssl_ciphers HIGH:!aNULL:!MD5;
    
    # oauth2-proxy endpoints
    location /oauth2/ {
        proxy_pass http://127.0.0.1:4180;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
	    proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        proxy_set_header X-Forwarded-Host $host;
    }

    # Auth check endpoint
    location = /oauth2/auth {
        internal;
	    proxy_pass http://127.0.0.1:4180;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        proxy_set_header X-Forwarded-Host $host;
	    proxy_set_header Content-Length "";
        proxy_pass_request_body off;
    }
    # Prefect UI
    location / {
        auth_request /oauth2/auth;
        error_page 401 = /oauth2/sign_in;
        proxy_set_header Authorization "Basic REDACTED";

        # Get IP address from docker inspect prefect-server-prefect-server-1
        proxy_pass http://172.18.0.6:4200;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        proxy_set_header X-Forwarded-Host $host;
    }
    # Prefect API (bypass oauth2-proxy)
    location /api/ {
        proxy_pass http://127.0.0.1:4200/api/;

        proxy_set_header Host $host;
        proxy_set_header Authorization $http_authorization;
    }
}
